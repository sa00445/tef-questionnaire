        // --- Checkbox Limiter Logic ---

        const personalityCheckboxes = document.querySelectorAll('input[name="personality"]');

        const maxPersonalityChecks = 3;

        function limitPersonalityChecks() {

            const checkedCount = document.querySelectorAll('input[name="personality"]:checked').length;

            personalityCheckboxes.forEach(checkbox => {

                if (checkedCount >= maxPersonalityChecks && !checkbox.checked) {

                    checkbox.disabled = true;

                    checkbox.nextElementSibling.classList.add('opacity-50', 'cursor-not-allowed');

                } else {

                    checkbox.disabled = false;

                    checkbox.nextElementSibling.classList.remove('opacity-50', 'cursor-not-allowed');

                }

            });

        }

        personalityCheckboxes.forEach(checkbox => {

            checkbox.addEventListener('change', limitPersonalityChecks);

        });

        function equalizePersonalityBoxHeights() {

        }
